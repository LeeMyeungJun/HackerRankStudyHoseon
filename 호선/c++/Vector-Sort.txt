#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

int main() 
{
    vector<int> v;
    int n;
    int vvvvv;

    cin >> n;

    for(int i = 0; i < n; i++)
    {
        cin >> vvvvv;
        v.push_back(vvvvv);
    }

    sort(v.begin(), v.end());

    for(int i = 0; i < v.size(); i++)
        cout << v[i] << " ";

    return 0;
}
